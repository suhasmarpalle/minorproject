create database bank_management;
use bank_management;
CREATE TABLE CUSTOMER
( NAME VARCHAR(100) NOT NULL DEFAULT 'INVALID INPUT',
  EMIAL_ID VARCHAR(100) NOT NULL DEFAULT 'INVALID INPUT',
  PHONE_NO VARCHAR(15) NOT NULL DEFAULT 'INVALID INPUT',
  USERNAME VARCHAR(30) NOT NULL DEFAULT 'INVALID INPUT',
  PASSWORD VARCHAR(20) NOT NULL DEFAULT 'INVALID INPUT',
  PRIMARY KEY(USERNAME));
  
  
  
  
  CREATE TABLE ACCOUNT
  ( ACCOUNT_ID VARCHAR(20) NOT NULL DEFAULT 'INVALID INPUT',
    NAME VARCHAR(100) NOT NULL DEFAULT 'INVALID INPUT',
    AGE INT NOT NULL DEFAULT 'INVALID INPUT',
    DOB DATE NOT NULL DEFAULT 'INVALID INPUT',
    MARITAL_STATUS VARCHAR(10) NOT NULL DEFAULT 'INVALID INPUT',
    EMAIL_ID VARCHAR(20) NOT NULL DEFAULT 'INVALID INPUT',
    PHONE_NO VARCHAR(15) NOT NULL DEFAULT 'INVALID INPUT',
    FATHER_NAME VARCHAR(100) NOT NULL DEFAULT 'INVALID INPUT',
    ADDRESS VARCHAR(50) NOT NULL DEFAULT 'INVALID INPUT',
    PINCODE VARCHAR(10) NOT NULL DEFAULT 'INVALID INPUT',
    CITY VARCHAR (25) NOT NULL DEFAULT 'INVALID INPUT',
    STATE VARCHAR(20) NOT NULL DEFAULT 'INVALID INPUT',
    COUNTRY VARCHAR(20) NOT NULL DEFAULT 'INVALID INPUT',
    PRIMARY KEY (ACCOUNT_ID));
    
    
 
 CREATE TABLE LOAN 
    ( LOAN_ID VARCHAR(15) NOT NULL DEFAULT 'INVALID INPUT',
	   ACCOUNT_HOLDER_NAME VARCHAR(100) NOT NULL DEFAULT  'INVALID INPUT',
       SANCTION_AMT  INT(25) NOT NULL DEFAULT 'INVALID INPUT',
       IOP INT(25) NOT NULL DEFAULT 'INVALID INPUT',
       ACCOUNT_ID VARCHAR(100) NOT NULL DEFAULT 'INVALID INPUT',
        PRIMARY KEY (LOAN_ID));
        
 CREATE TABLE TRANSCATION
     ( TRANSACTION_ID VARCHAR(100) NOT NULL DEFAULT 'INVALID INPUT',
     FROM_BANKACC  VARCHAR(100) NOT NULL DEFAULT 'INVALID INPUT',
     TO_BANKACC  VARCHAR(100) NOT NULL DEFAULT 'INVALID INPUT',
     TRANSACTION_AMOUNT INT(25) NOT NULL DEFAULT 'INVALID INPUT',
     PRIMARY KEY(TRANSACTION_ID));
     
CREATE TABLE FIX_DEPOSIT
( 
FD_ID  VARCHAR(100) NOT NULL DEFAULT 'INVALID INPUT',
ACCOUNT_NAME  VARCHAR(100) NOT NULL DEFAULT 'INVALID INPUT',
AMOUNT INT(25)  NOT NULL DEFAULT 'INVALID INPUT',
MATURITY_PERIOD INT (25)  NOT NULL DEFAULT 'INVALID INPUT'
);

CREATE TABLE FME(
 FME_ID VARCHAR(15) NOT NULL DEFAULT 'INVALID INPUT',
 F_AMT VARCHAR(30) NOT NULL DEFAULT 'INVALID INPUT',
 I_AMT VARCHAR(30) NOT NULL DEFAULT 'INVALID INPUT',
 PRIMARY KEY (FME_ID)
 );
 
 
 
CREATE TABLE ADMIN 
( 
  USERNAME VARCHAR(45) NOT NULL DEFAULT 'INVALID INPUT',
  PASSWORD VARCHAR(45) NOT NULL DEFAULT 'INVALID INPUT',
  CUSTOMER_ID VARCHAR(45) references CUSTOMER(USERNAME),
  ACCOUNTID VARCHAR(45) references ACCOUNT(ACCOUNT_ID),
  LOANID VARCHAR(45) REFERENCES LOAN(LOAN_id),
  FMEID VARCHAR(45) references FME(FME_ID),
  TRANSACTIONID VARCHAR(45) references TRANSACTION(TRANSACTION_ID),
  FDID VARCHAR(45) references FIX_DEPOSIT(FD_ID)
  );
